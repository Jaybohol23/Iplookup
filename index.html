OLYMPIC RINGS TURTLE


import turtle
# draw a circle with position
def draw_circle(color, x, y):
    turtle.penup()
    turtle.goto(x, y)
    turtle.pendown()
    turtle.color(color)
    turtle.circle(50)

# Set up the screen
turtle.bgcolor("white")
turtle.speed(5)

# Draw the Olympic rings
draw_circle("blue", -120, 0)
draw_circle("black", 0, 0)
draw_circle("red", 120, 0)
draw_circle("yellow", -60, -50)
draw_circle("green", 60, -50)

turtle.hideturtle()
turtle.done()
